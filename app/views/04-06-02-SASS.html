Why ?

  • Run on server side
  • Compiled Languages
  • Extends CSS languages
  • More simple
  • Running on Grunt server
@import 'reset';

$font-stack:    Helvetica, sans-serif;
$primary-color: #333;

body {
    font: 100% $font-stack;
    color: $primary-color;
}
@mixin border-radius($radius) {
    -webkit-border-radius: $radius;
       -moz-border-radius: $radius;
        -ms-border-radius: $radius;
         -o-border-radius: $radius;
            border-radius: $radius;
}

.box { @include border-radius(10px); }

.box-success {
    @extend .box;
    border-color: green;
}
