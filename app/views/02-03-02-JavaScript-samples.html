Closure

var closure = function() {
    var count = 0;
    return function () {
        return ++count;
    };
}
var inc = closure();
inc(); // returns 1
inc(); // returns 2
inc(); // returns 3

Variadic function

var sum = function() {
var i, x = 0;
for (i = 0; i < arguments.length; ++i) {
    x += arguments[i];
}
    return x;
}
sum(1, 2, 3); // returns 6
